    <script>
        function validate() {
            var regno = document.getElementById('regno').value;
            var test1 = parseInt(document.getElementById('test1').value);
            var test2 = parseInt(document.getElementById('test2').value);
            var test3 = parseInt(document.getElementById('test3').value);
            var regnoregex = /^\d{2}[A-Za-z]{3}\d{2}[A-Za-z]{2}\d{3}$/;

            if (!regnoregex.test(regno)) {
                alert('Give correct Regno.');
                return;
            }

            if (!validateMarks(test1) || !validateMarks(test1) || !validateMarks(test1)) {
                alert('Enter the marks(0-20)');
                return;
            }

            var avg = (test1 + test2 + test3 - Math.min(test1, test2, test3)) / 2;
            alert('Average = ' + avg.toFixed(2));
        }
        function validateMarks(marks) {
            return !isNaN(marks) && marks >= 0 && marks <= 20;
        }
    </script>
